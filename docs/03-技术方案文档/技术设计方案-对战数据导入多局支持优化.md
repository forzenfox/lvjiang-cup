## 一、概述

### 1.1 改造目标

将当前**单文件单局**导入模式升级为**单文件多局**模式，核心变更：

1. Excel 模板从单 Sheet 改为多 Sheet（每个 Sheet 对应一局对局）
2. Sheet 名称标准化（`第X局`），自动解析局数
3. 新增局数校验（一致性、唯一性、赛制合法性）
4. 模板根据比分动态生成，不再缓存
5. 不兼容旧版单 Sheet 模板

### 1.2 不涉及变更

- 数据库表结构（`match_games`、`player_match_stats` 无变更）
- 前端对战数据展示页面
- 选手匹配逻辑

```typescript
export function parseSheetGameNumber(sheetName: string): number | null {
  const trimmed = sheetName.trim();
  const match = trimmed.match(SHEET_NAME_REGEX);
  if (!match) return null;

  const numStr = match[1];
  if (CHINESE_NUMBER_MAP[numStr] !== undefined) {
    return CHINESE_NUMBER_MAP[numStr];
  }
  return parseInt(numStr, 10);
}
```

```typescript
// 校验多个Sheet的局数唯一性
export function validateGameNumberUniqueness(
  sheets: ValidSheetInfo[],
): SheetValidationResult {
  const errors: string[] = [];
  const numberMap = new Map<number, string[]>();

  for (const item of sheets) {
    if (!numberMap.has(item.gameNumber)) {
      numberMap.set(item.gameNumber, []);
    }
    numberMap.get(item.gameNumber)!.push(item.sheetName);
  }

  for (const [gameNumber, sheetNames] of numberMap) {
    if (sheetNames.length > 1) {
      errors.push(
        `Sheet「${sheetNames.join('」和Sheet「')}」解析的局数均为${gameNumber}，局数不能重复`,
      );
    }
  }

  return { valid: errors.length === 0, errors };
}
```

#### 2.1.4 赛制合法性校验函数

```typescript
// 校验局数是否在赛制允许范围内
export function validateGameNumberForFormat(
  gameNumber: number,
  boFormat: string,
): boolean {
  const maxGames = BO_FORMAT_MAX_GAMES[boFormat] || 1;
  return gameNumber >= 1 && gameNumber <= maxGames;
}

// 批量校验多个Sheet的赛制合法性
export function validateSheetsForFormat(
  sheets: ValidSheetInfo[],
  boFormat: string,
): SheetValidationResult {
  const errors: string[] = [];
  const maxGames = BO_FORMAT_MAX_GAMES[boFormat] || 1;

  for (const sheet of sheets) {
    if (!validateGameNumberForFormat(sheet.gameNumber, boFormat)) {
      errors.push(
        `Sheet「${sheet.sheetName}」解析的局数为${sheet.gameNumber}，但当前对战的赛制为${boFormat}，最多允许${maxGames}局`,
      );
    }
  }

  return { valid: errors.length === 0, errors };
}
```

#### 2.1.5 比分校验函数

```typescript
// 根据比分计算模板Sheet数量
export function calculateSheetCount(scoreA: number, scoreB: number): number {
  return scoreA + scoreB;
}

// 校验比分是否满足赛制约束
export function validateScoreForFormat(
  scoreA: number,
  scoreB: number,
  boFormat: string,
): SheetValidationResult {
  const errors: string[] = [];
  const maxGames = BO_FORMAT_MAX_GAMES[boFormat] || 1;
  const winScore = BO_FORMAT_WIN_SCORE[boFormat] || 1;

  const totalGames = scoreA + scoreB;
  const maxScore = Math.max(scoreA, scoreB);
  const minScore = Math.min(scoreA, scoreB);

  if (totalGames < 1 || totalGames > maxGames) {
    errors.push(`比分 ${scoreA}:${scoreB} 的总场数不在1-${maxGames}范围内`);
  }

  if (maxScore < winScore) {
    errors.push(`${boFormat}赛制中胜方至少需要${winScore}分，当前最高分为${maxScore}`);
  }

  if (minScore >= winScore) {
    errors.push(`双方得分均达到${winScore}，比分不合法`);
  }

  return { valid: errors.length === 0, errors };
}
```

#### 2.1.6 局数一致性校验函数

```typescript
// 校验Sheet局数与MatchInfo局数的一致性
export function validateGameNumberConsistency(
  sheetGameNumber: number,
  excelGameNumber: number,
): { consistent: boolean; warning: GameNumberWarning | null } {
  // MatchInfo局数为空或0，以Sheet局数为准，不告警
  if (!excelGameNumber || excelGameNumber === 0) {
    return { consistent: true, warning: null };
  }

  // 一致
  if (sheetGameNumber === excelGameNumber) {
    return { consistent: true, warning: null };
  }

  // 不一致，生成告警
  return {
    consistent: false,
    warning: {
      sheetName: '', // 由调用方填充
      sheetGameNumber,
      excelGameNumber,
      resolvedGameNumber: sheetGameNumber,
      message: `Sheet名称解析的局数为${sheetGameNumber}，但表格中填写的局数为${excelGameNumber}，将以Sheet名称的局数${sheetGameNumber}为准`,
    },
  };
}
```

#### 2.2.2 导入流程改造

核心流程变更：从解析单个 Sheet 改为遍历多个 Sheet。

```typescript
async importMatchData(
  matchId: string,
  file: Express.Multer.File,
  adminId: string,
  options: ImportOptions = {},
): Promise<MultiGameImportResponse> {
  const { dryRun = false, confirmWarnings = false } = options;

  // 1. 读取Excel文件
  const workbook = xlsx.read(file.buffer, { type: 'buffer' });

  // 2. 解析所有Sheet名称，提取有效Sheet
  const validSheets: ValidSheetInfo[] = [];
  for (const sheetName of workbook.SheetNames) {
    const gameNumber = parseSheetGameNumber(sheetName);
    if (gameNumber !== null) {
      validSheets.push({ sheetName, gameNumber });
    }
  }

  // 3. 校验：至少一个有效Sheet
  if (validSheets.length === 0) {
    throw new HttpException(
      '未找到有效的Sheet，Sheet名称须为"第X局"格式',
      40013,
    );
  }

  // 4. 校验：局数唯一性
  const uniquenessResult = validateGameNumberUniqueness(validSheets);
  if (!uniquenessResult.valid) {
    throw new HttpException({ code: 40012, errors: uniquenessResult.errors }, 400);
  }

  // 5. 校验：赛制合法性
  const match = await this.getMatchInfo(matchId);
  const formatResult = validateSheetsForFormat(validSheets, match.bo_format);
  if (!formatResult.valid) {
    throw new HttpException({ code: 40011, errors: formatResult.errors }, 400);
  }

  // 6. 按局数排序后逐Sheet解析数据
  const sortedSheets = validSheets.sort((a, b) => a.gameNumber - b.gameNumber);
  const warnings: GameNumberWarning[] = [];
  const parsedResults: ParsedMatchData[] = [];

  for (const sheetInfo of sortedSheets) {
    const sheet = workbook.Sheets[sheetInfo.sheetName];
    const parsedData = parseMatchDataExcel(sheet);

    // 7. 局数一致性校验
    const consistency = validateGameNumberConsistency(
      sheetInfo.gameNumber,
      parsedData.matchInfo.gameNumber,
    );

    if (!consistency.consistent && consistency.warning) {
      const warning = { ...consistency.warning, sheetName: sheetInfo.sheetName };
      warnings.push(warning);
    }

    // 以Sheet局数为准
    parsedData.matchInfo.gameNumber = sheetInfo.gameNumber;
    parsedResults.push(parsedData);
  }

  // 8. 如果有告警且用户未确认，返回告警
  if (warnings.length > 0 && !confirmWarnings) {
    return {
      imported: false,
      totalGames: sortedSheets.length,
      results: [],
      warnings,
    } as MultiGameImportWarningResponse;
  }

  // 9. 预检模式，不执行导入
  if (dryRun) {
    return {
      imported: false,
      totalGames: sortedSheets.length,
      results: sortedSheets.map(s => ({
        gameNumber: s.gameNumber,
        imported: false,
        playerCount: 0,
        failedCount: 0,
        overwritten: false,
      })),
    };
  }

  // 10. 逐Sheet导入数据（每局独立事务）
  const results = [];
  for (const parsedData of parsedResults) {
    try {
      const result = await this.importSingleGameData(matchId, parsedData, adminId);
      results.push(result);
    } catch (error) {
      results.push({
        gameNumber: parsedData.matchInfo.gameNumber,
        imported: false,
        playerCount: 0,
        failedCount: 0,
        overwritten: false,
        error: error.message,
      });
    }
  }

  return {
    imported: results.every(r => r.imported),
    totalGames: results.length,
    results,
  };
}
```

```typescript
async generateTemplate(matchId: string): Promise<Buffer> {
  // 1. 查询对战信息
  const match = await this.databaseService.get<any>(
    `SELECT m.id, m.bo_format, m.score_a, m.score_b, m.status,
            ta.name as team_a_name, tb.name as team_b_name
     FROM matches m
     LEFT JOIN teams ta ON m.team_a_id = ta.id
     LEFT JOIN teams tb ON m.team_b_id = tb.id
     WHERE m.id = ?`,
    [matchId],
  );

  if (!match) {
    throw new HttpException('对战不存在', 404);
  }

  if (match.status !== 'finished') {
    throw new HttpException('对战尚未结束，无法下载模板', 40020);
  }

  // 2. 校验比分
  const scoreResult = validateScoreForFormat(match.score_a, match.score_b, match.bo_format);
  if (!scoreResult.valid) {
    throw new HttpException({ code: 40021, errors: scoreResult.errors }, 400);
  }

  // 3. 计算Sheet数量
  const sheetCount = calculateSheetCount(match.score_a, match.score_b);

  // 4. 生成多Sheet的Excel文件
  const workbook = new ExcelJS.Workbook();

  for (let i = 1; i <= sheetCount; i++) {
    const sheetName = `第${NUMBER_TO_CHINESE[i]}局`; // 第一局、第二局...
    await this.createMatchDataSheet(workbook, sheetName, {
      redTeamName: match.team_a_name,
      blueTeamName: match.team_b_name,
      gameNumber: i,
    });
  }

  // 5. 生成Buffer返回
  const buffer = await workbook.xlsx.writeBuffer() as Buffer;
  return buffer;
}
```

```typescript
@Get(':matchId/import/template')
@ApiOperation({ summary: '下载对战数据导入模板（根据比分动态生成）' })
async downloadTemplate(
  @Param('matchId') matchId: string,
  @Res() res: Response,
) {
  const buffer = await this.matchDataImportService.generateTemplate(matchId);

  // 从Buffer中获取对战信息用于文件名
  // 需要在service中返回文件名信息
  const match = await this.matchDataService.getMatchBasicInfo(matchId);
  const fileName = `驴酱杯对战信息_${match.team_a_name}_vs_${match.team_b_name}_${match.bo_format}.xlsx`;

  res.setHeader(
    'Content-Type',
    'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
  );
  res.setHeader(
    'Content-Disposition',
    `attachment; filename*=UTF-8''${encodeURIComponent(fileName)}`,
  );
  res.setHeader('Content-Length', buffer.length.toString());
  res.send(buffer);
}
```

#### 2.4.3 废弃 API

以下 API 将被移除：

| API | 说明 |
|-----|------|
| `GET /admin/matches/import/template` | 旧版通用模板下载，替换为 `GET /admin/matches/:matchId/import/template` |
| `GET /admin/matches/import/template/refresh` | 模板刷新，不再需要（模板不再缓存） |

**路由冲突处理**：由于 `import` 可能被当作 `:matchId`，需要确保 `:matchId/import/template` 路由在控制器中优先匹配。建议在控制器中将新路由定义在旧路由之前，或使用路由守卫进行区分。最终方案：**直接移除旧路由**，避免冲突。

```typescript
// 导入对战数据（支持多Sheet）
export async function importMatchData(
  matchId: string,
  file: File,
  options?: ImportOptions,
): Promise<MultiGameImportResponse> {
  const formData = new FormData();
  formData.append('file', file);
  if (options?.dryRun) formData.append('dryRun', 'true');
  if (options?.confirmWarnings) formData.append('confirmWarnings', 'true');

  const responseData = await post<ApiResponse<MultiGameImportResponse>>(
    `/admin/matches/${matchId}/games/import`,
    formData,
  );

  if (!responseData.success || !responseData.data) {
    throw new Error(responseData.message || '导入对战数据失败');
  }

  return responseData.data;
}
```

```typescript
const handleImport = async () => {
  if (!selectedFile) return;

  try {
    setImporting(true);

    // 第一步：预检
    const preCheckResult = await importMatchData(matchId, selectedFile, {
      dryRun: true,
    });

    // 如果有局数不一致告警
    if (preCheckResult.warnings && preCheckResult.warnings.length > 0) {

      setShowWarningDialog(true);
      setImporting(false);
      return;
    }

    // 无告警，直接导入
    await doImport(false);
  } catch (error: any) {
    // 校验失败，后台已清除错误文件
    setSelectedFile(null);
    setError(error.message);
  } finally {
    setImporting(false);
  }
};

// 确认告警后继续导入

  setShowWarningDialog(false);
  await doImport(true);
};

const doImport = async (confirmWarnings: boolean) => {
  if (!selectedFile) return;

  try {
    setImporting(true);
    const result = await importMatchData(matchId, selectedFile, {
      confirmWarnings,
    });

    // 导入成功，清除文件
    setSelectedFile(null);
    setImportResult(result);
    onImportSuccess?.(result);
  } catch (error: any) {
    // 导入失败，清除文件
    setSelectedFile(null);
    setError(error.message);
  } finally {
    setImporting(false);
  }
};

// 取消告警
const handleCancelWarnings = () => {
  setShowWarningDialog(false);
  // 清除文件
  setSelectedFile(null);
  setWarnings([]);
};

  setWarnings([]);
  setError(null);
```

## 五、校验逻辑设计

### 5.1 校验执行顺序

```
Sheet名称格式校验 → 局数唯一性校验 → 赛制合法性校验
    → 逐Sheet数据解析 → 局数一致性校验 → 战队名匹配校验
    → 数据字段校验 → 英雄名称校验 → 选手匹配校验
```

### 5.2 校验失败处理策略

| 校验阶段 | 失败策略 | 错误码 |
|----------|---------|--------|
| Sheet名称格式 | 全部终止，返回错误 | 40013 |
| 局数唯一性 | 全部终止，返回错误 | 40012 |
| 赛制合法性 | 全部终止，返回错误 | 40011 |
| 局数一致性 | 返回告警，等待用户确认 | 40010 |
| 战队名匹配 | 该局终止，其他局继续 | 40001 |
| 数据字段校验 | 该局终止，其他局继续 | 40001 |
| 选手匹配 | 该局部分成功，记录失败选手 | 40001 |

### 5.3 事务策略

- 每局导入使用独立事务（`BEGIN` → `INSERT/UPDATE` → `COMMIT`）
- 某局失败不影响其他已成功导入的局
- 前端根据 `results` 中每局的 `imported` 状态展示结果
