#### 3.2.4 WeChatSection 组件（核心交互）

**文件**: `src/components/layout/Footer/WeChatSection.tsx`

```typescript
import React, { useState, useEffect, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';

interface WeChatSectionProps {
  name: string;
  qrCode: string;
  size: number;
}

export const WeChatSection: React.FC<WeChatSectionProps> = ({
  name,
  qrCode,
  size
}) => {
  const [isHovered, setIsHovered] = useState(false);
  const [displaySrc, setDisplaySrc] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [hasError, setHasError] = useState(false);
  const hasAttemptedLoad = useRef(false);

  // 图片加载逻辑：CDN 优先，失败则降级到本地
  const loadImage = (src: string, isRetry: boolean = false) => {
    const img = new Image();
    img.onload = () => {
      setDisplaySrc(src);
      setIsLoading(false);
    };
    img.onerror = () => {
      // CDN 加载失败，如果是第一次尝试，则降级到本地路径
      if (!isRetry) {
        // 尝试本地路径兜底
        const localPath = qrCode;
        if (localPath !== src) {
          loadImage(localPath, true);
          return;
        }
      }
      // 两者都失败
      setHasError(true);
      setIsLoading(false);
    };
    img.src = src;
  };

  // 首次悬停时加载图片
  useEffect(() => {
    if (isHovered && !hasAttemptedLoad.current && !displaySrc && !hasError) {
      hasAttemptedLoad.current = true;
      // 优先从 CDN 加载
      loadImage(qrCode);
    }
  }, [isHovered]);

  return (
    <div 
      className="relative flex flex-col items-center"
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
    >
      {/* 二维码浮层（悬停时显示） */}
      <AnimatePresence>
        {isHovered && (
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: 10 }}
            transition={{ duration: 0.2, ease: 'easeInOut' }}
            className="absolute bottom-full mb-4 z-50"
          >
            <div className="relative">
              {/* 加载状态 */}
              {isLoading && (
                <div 
                  className="bg-white rounded-lg shadow-lg p-2"
                  style={{ width: size, height: size }}
                >
                  <div className="w-full h-full flex items-center justify-center">
                    <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-gold" />
                  </div>
                </div>
              )}
              
              {/* 加载成功 */}
              {!isLoading && displaySrc && (
                <>
                  <img
                    src={displaySrc}
                    alt="微信公众号二维码"
                    style={{ width: size, height: size }}
                    className="rounded-lg shadow-lg bg-white p-2"
                  />
                  {/* 小三角指示器 */}
                  <div className="absolute -bottom-2 left-1/2 transform -translate-x-1/2 w-0 h-0 border-l-8 border-r-8 border-t-8 border-l-transparent border-r-transparent border-t-white" />
                </>
              )}
              
              {/* 加载失败 */}
              {!isLoading && hasError && (
                <div 
                  className="bg-white rounded-lg shadow-lg p-2 border-2 border-red-300"
                  style={{ width: size, height: size }}
                >
                  <div className="w-full h-full flex items-center justify-center text-gray-400 text-xs text-center px-2">
                    二维码加载失败
                  </div>
                </div>
              )}
            </div>
          </motion.div>
        )}
      </AnimatePresence>
      
      {/* 微信公众号文字 */}
      <div className="text-sm text-white hover:text-gold cursor-pointer">
        <span className="hover:underline">微信公众号：{name}</span>
      </div>
    </div>
  );
};
```

**核心特性**：
1. **CDN 优先策略**：优先从 CDN 地址加载图片（如果配置了 CDN）
2. **自动降级兜底**：CDN 失败自动切换到本地路径
3. **懒加载优化**：仅在首次悬停时加载图片，避免不必要的请求
4. **加载状态展示**：显示 loading 旋转动画
5. **错误处理**：加载失败显示友好提示，红色边框标识
6. **防重复加载**：使用 `useRef` 避免重复尝试加载

### 4.3 资源加载策略

**微信公众号二维码加载逻辑**（与封面图片一致）：

```typescript
// 1. 优先尝试 CDN 加载
const cdnSrc = `${GITHUB_CDN_BASE}/assets/lvlvdianjing.webp`;

// 2. CDN 失败自动降级到本地路径
const localSrc = '/assets/lvlvdianjing.webp';

// 3. 组件内部实现
const loadImage = (src: string, isRetry: boolean = false) => {
  const img = new Image();
  img.onload = () => {
    // 加载成功，显示图片
    setDisplaySrc(src);
    setIsLoading(false);
  };
  img.onerror = () => {
    // CDN 失败，降级到本地
    if (!isRetry && src === cdnSrc) {
      loadImage(localSrc, true);
      return;
    }
    // 两者都失败，显示错误
    setHasError(true);
    setIsLoading(false);
  };
  img.src = src;
};
```

### 4.4 懒加载优化

```typescript
// 仅在首次悬停时加载图片
useEffect(() => {
```

**优化效果**：
- ✅ 首次悬停才加载，避免不必要的请求
- ✅ 加载成功后缓存，后续悬停直接显示
- ✅ CDN 失败自动降级，提升可靠性
- ✅ 与封面图片加载策略一致，统一技术栈

## 10. 常见问题

### Q1: 二维码图片加载失败怎么办？

**A**: 检查以下几点：
1. 确认文件路径正确：`/assets/lvlvdianjing.webp`
2. 确认文件存在于 `public/assets/` 目录
3. 检查浏览器控制台是否有 404 错误
4. 添加错误处理：
```typescript
<img 
  src={qrCode} 
  alt="微信公众号二维码"
  onError={(e) => {
    (e.target as HTMLImageElement).src = '/fallback-qr.png';
  }}
/>
```

### Q2: 移动端也显示了页脚？

**A**: 检查响应式类名：
```typescript
<footer className="hidden lg:block">  // 正确
// lg 断点为 1024px，移动端和平板端会隐藏
```

### Q3: 悬停动画不流畅？

**A**: 优化动画配置：
```typescript
<motion.div
  transition={{ 
    duration: 0.2, 
    ease: 'easeInOut'  // 使用缓动函数
  }}
>
```
