## 3. 数据库设计

### 3.1 视频数据表（videos）

```sql
CREATE TABLE IF NOT EXISTS videos (
  id TEXT PRIMARY KEY,
  bvid TEXT NOT NULL,
  page INTEGER DEFAULT 1,
  bilibili_title TEXT,           -- B站原始标题（API获取）
  custom_title TEXT,             -- 自定义标题（可选）
  cover_url TEXT,                -- B站封面URL（API获取）
  "order" INTEGER DEFAULT 0,
  status INTEGER DEFAULT 1,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  created_by TEXT NOT NULL,
  UNIQUE(bvid, page)
);

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_videos_order_status ON videos("order", status);
CREATE INDEX IF NOT EXISTS idx_videos_status ON videos(status);
```

### 3.2 字段说明

| 字段 | 类型 | 必填 | 默认值 | 说明 |
|------|------|------|--------|------|
| id | TEXT | 是 | - | UUID 主键 |
| bvid | TEXT | 是 | - | B 站视频 ID |
| page | INTEGER | 否 | 1 | 分 P 页码 |
| bilibili_title | TEXT | 否 | NULL | B站原始标题（API获取） |
| custom_title | TEXT | 否 | NULL | 自定义标题（优先级高于bilibili_title） |
| cover_url | TEXT | 否 | NULL | B站封面图 URL（API获取） |
| order | INTEGER | 是 | 0 | 排序值，越小越靠前 |
| status | INTEGER | 是 | 1 | 状态：0-禁用，1-启用 |

> **标题显示逻辑**：`custom_title || bilibili_title || '未命名视频'`
| created_at | DATETIME | 是 | CURRENT_TIMESTAMP | 创建时间 |
| updated_at | DATETIME | 是 | CURRENT_TIMESTAMP | 更新时间 |
| created_by | TEXT | 是 | - | 创建人（管理员 ID） |

**实现逻辑**：
```sql
SELECT
  id,
  COALESCE(custom_title, bilibili_title, '未命名视频') as title,
  bvid,
  page,
  cover_url
FROM videos
WHERE status = 1
ORDER BY "order" ASC, created_at DESC
LIMIT 10;
```

**封面直连说明**：
- B站封面URL（i0/i1/i2.hdslb.com）可直接在前端 `<img>` 标签中使用
- `<img>`、CSS `background-image` 等HTML标签加载跨域资源不受CORS限制
- 只需在CSP中配置允许B站图片域名即可

function extractBvidFromUrl(url: string): { bvid: string | null; page: number } {
  const patterns = [
    /bilibili\.com\/video\/(BV[a-zA-Z0-9]+)/,
    /bilibili\.com\/video\/(BV[a-zA-Z0-9]+)\?p=(\d+)/,
    /bilibili\.com\/video\/(BV[a-zA-Z0-9]+)\//,

    if (match) {
      return { bvid: match[1], page: match[2] ? parseInt(match[2]) : 1 };

  }

}

### 5.2 服务层实现

#### 5.2.1 视频服务

```typescript
// src/services/videoService.ts
import axios from 'axios';
import type { Video, VideoItem } from '@/types';

const API_BASE = '/api';

export const videoService = {
  // 获取前端视频列表
  async getFrontendVideos(): Promise<{ list: VideoItem[]; total: number }> {
    const response = await axios.get(`${API_BASE}/videos`);
    return response.data.data;
  },

  // 获取后台视频列表
  async getBackendVideos(params?: {
    page?: number;
    pageSize?: number;
    keyword?: string;
    status?: 0 | 1;
  }): Promise<{ list: Video[]; total: number; page: number; pageSize: number }> {
    const response = await axios.get(`${API_BASE}/admin/videos`, { params });
    return response.data.data;
  },

  // 添加视频
  async createVideo(data: {
    title: string;
    bvid: string;
    page?: number;
    coverUrl?: string;
    order?: number;
  }): Promise<{ id: string }> {
    const response = await axios.post(`${API_BASE}/admin/videos`, data);
    return response.data.data;
  },

  // 更新视频
  async updateVideo(
    id: string,
    data: {
      title?: string;
      bvid?: string;
      page?: number;
      coverUrl?: string;
      order?: number;
      status?: 0 | 1;
    },
  ): Promise<Video> {
    const response = await axios.put(`${API_BASE}/admin/videos/${id}`, data);
    return response.data.data;
  },

  // 删除视频
  async deleteVideo(id: string): Promise<void> {
    await axios.delete(`${API_BASE}/admin/videos/${id}`);
  },

  // 批量排序
  async batchSort(ids: string[]): Promise<void> {
    await axios.put(`${API_BASE}/admin/videos/sort`, { ids });
  },
};
```

### 5.3 核心组件实现

#### 5.3.1 VideoCarousel 主组件

```tsx
import React, { useState, useEffect, useCallback } from 'react';
import { VideoPlayer } from './VideoPlayer';
import { VideoThumbnail } from './VideoThumbnail';
import { ControlArrows } from './ControlArrows';
import { Indicator } from './Indicator';
import { useAutoplay } from './hooks/useAutoplay';
import { useSwipe } from './hooks/useSwipe';
import { useMediaQuery } from '@/hooks/useMediaQuery';

interface VideoCarouselProps {
  videos: VideoItem[];
}

export const VideoCarousel: React.FC<VideoCarouselProps> = ({ videos }) => {
  const [currentIndex, setCurrentIndex] = useState(0);
  const isDesktop = useMediaQuery('(min-width: 768px)');
  
  // 自动轮播 Hook
  const { startAutoplay, stopAutoplay, resetTimer } = useAutoplay({
    interval: 5000,
    pauseDuration: 30000,
    onAutoplay: handleNext,
    enabled: isDesktop && videos.length > 2,
  });
  
  // 触摸滑动 Hook
  const { swipeProps } = useSwipe({
    onSwipeLeft: handleNext,
    onSwipeRight: handlePrev,
    threshold: 50,
  });
  
  // 切换到指定视频
  const goToIndex = useCallback(
    (index: number) => {
      let newIndex = index;
      if (newIndex < 0) newIndex = videos.length - 1;
      if (newIndex >= videos.length) newIndex = 0;

      setCurrentIndex(newIndex);
      stopAutoplay();
      resetTimer();
    },
    [videos.length],
  );
  
  const handlePrev = useCallback(() => {
    goToIndex(currentIndex - 1);
  }, [currentIndex, goToIndex]);
  
  const handleNext = useCallback(() => {
    goToIndex(currentIndex + 1);
  }, [currentIndex, goToIndex]);
  
  // 视频数量 ≤ 2 时隐藏控制元素
  const showControls = videos.length > 2;
  
  return (
    <div className="video-carousel w-full py-5" {...swipeProps}>
      <div className="carousel-container flex items-center justify-center gap-5">
        {/* 左侧缩略图 */}
        {showControls && (
          <VideoThumbnail
            video={videos[getPrevIndex()]}
            position="left"
            onClick={handlePrev}
          />
        )}

        {/* 主视频播放器 */}
        <VideoPlayer
          video={videos[currentIndex]}
          autoplay={isDesktop}
          muted={isDesktop}
        />

        {/* 右侧缩略图 */}
        {showControls && (
          <VideoThumbnail
            video={videos[getNextIndex()]}
            position="right"
            onClick={handleNext}
          />
        )}

        {/* 控制箭头 */}
        {showControls && (
          <ControlArrows onPrev={handlePrev} onNext={handleNext} />
        )}
      </div>

      {/* 滚动指示器 */}
      <Indicator
        current={currentIndex + 1}
        total={videos.length}
        showDots={!isDesktop}
      />
    </div>
  );
};
```

### 6.4 服务层实现

```typescript
// videos.service.ts
import {
  Injectable,
  Logger,
  NotFoundException,
  ConflictException,
  BadRequestException,
} from '@nestjs/common';
import { DatabaseService } from '../../database/database.service';
import { CacheService } from '../../cache/cache.service';
import { CreateVideoDto } from './dto/create-video.dto';
import { UpdateVideoDto } from './dto/update-video.dto';
import { Video } from './entities/video.entity';

@Injectable()
export class VideosService {
  private readonly logger = new Logger(VideosService.name);
  private readonly CACHE_KEY = 'videos:frontend';
  private readonly CACHE_KEY_ALL = 'videos:backend';

  constructor(
    private databaseService: DatabaseService,
    private cacheService: CacheService,
  ) {}

  // 获取前端视频列表
  async findFrontendVideos(): Promise<{ list: Video[]; total: number }> {
    const cached = this.cacheService.get(this.CACHE_KEY);
    if (cached) {
      return cached;
    }

    const results = await this.databaseService.all<any>(
      `SELECT id, title, bvid, page, cover_url as "coverUrl"
       FROM videos
       WHERE status = 1
       ORDER BY "order" ASC, created_at DESC
       LIMIT 10`,
    );

    const list = results.map((row) => this.mapToVideo(row));
    const total = list.length;

    const result = { list, total };
    this.cacheService.set(this.CACHE_KEY, result);

    return result;
  }

  // 获取后台视频列表
  async findBackendVideos(params: {
    page?: number;
    pageSize?: number;
    keyword?: string;
    status?: 0 | 1;
  }): Promise<{ list: Video[]; total: number; page: number; pageSize: number }> {
    const { page = 1, pageSize = 20, keyword, status } = params;
    const offset = (page - 1) * pageSize;

    let whereClause = '1=1';
    const values: any[] = [];

    if (keyword) {
      whereClause += ' AND title LIKE ?';
      values.push(`%${keyword}%`);
    }

    if (status !== undefined) {
      whereClause += ' AND status = ?';
      values.push(status);
    }

    const countResult = await this.databaseService.get<{ total: number }>(
      `SELECT COUNT(*) as total FROM videos WHERE ${whereClause}`,
      values,
    );

    const total = countResult?.total || 0;

    const results = await this.databaseService.all<any>(
      `SELECT * FROM videos WHERE ${whereClause}
       ORDER BY "order" ASC, created_at DESC
       LIMIT ? OFFSET ?`,
      [...values, pageSize, offset],
    );

    const list = results.map((row) => this.mapToVideo(row));

    return { list, total, page, pageSize };
  }

  // 创建视频
  async create(dto: CreateVideoDto, createdBy: string): Promise<{ id: string }> {
    // 验证 B 站视频
    await this.validateBilibiliVideo(dto.bvid);

    // 检查重复
    const exists = await this.databaseService.get<any>(
      'SELECT id FROM videos WHERE bvid = ? AND page = ?',
      [dto.bvid, dto.page || 1],
    );

    if (exists) {
      throw new ConflictException('视频已存在');
    }

    // 检查启用数量上限
    if (dto.status !== 0) {
      const enabledCount = await this.databaseService.get<{ count: number }>(
        'SELECT COUNT(*) as count FROM videos WHERE status = 1',
      );

      if (enabledCount.count >= 10) {
        throw new BadRequestException('最多可启用 10 个视频');
      }
    }

    const id = crypto.randomUUID();
    await this.databaseService.run(
      `INSERT INTO videos (id, title, bvid, page, cover_url, "order", status, created_by)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        id,
        dto.title,
        dto.bvid,
        dto.page || 1,
        dto.coverUrl,
        dto.order || 0,
        dto.status !== undefined ? dto.status : 1,
        createdBy,
      ],
    );

    this.clearCache();
    this.logger.log(`Video created: ${id}`);

    return { id };
  }

  // 更新视频
  async update(id: string, dto: UpdateVideoDto): Promise<Video> {
    const existing = await this.findById(id);
    if (!existing) {
      throw new NotFoundException('视频不存在');
    }

    // 如果更新 bvid，验证并检查重复
    if (dto.bvid && dto.bvid !== existing.bvid) {
      await this.validateBilibiliVideo(dto.bvid);

      const exists = await this.databaseService.get<any>(
        'SELECT id FROM videos WHERE bvid = ? AND page = ? AND id != ?',
        [dto.bvid, dto.page || existing.page, id],
      );

      if (exists) {
        throw new ConflictException('视频已存在');
      }
    }

    // 如果启用视频，检查数量上限
    if (dto.status === 1 && existing.status === 0) {
      const enabledCount = await this.databaseService.get<{ count: number }>(
        'SELECT COUNT(*) as count FROM videos WHERE status = 1',
      );

      if (enabledCount.count >= 10) {
        throw new BadRequestException('最多可启用 10 个视频');
      }
    }

    const updates: string[] = [];
    const values: any[] = [];

    if (dto.title !== undefined) {
      updates.push('title = ?');
      values.push(dto.title);
    }
    if (dto.bvid !== undefined) {
      updates.push('bvid = ?');
      values.push(dto.bvid);
    }
    if (dto.page !== undefined) {
      updates.push('page = ?');
      values.push(dto.page);
    }
    if (dto.coverUrl !== undefined) {
      updates.push('cover_url = ?');
      values.push(dto.coverUrl);
    }
    if (dto.order !== undefined) {
      updates.push('"order" = ?');
      values.push(dto.order);
    }
    if (dto.status !== undefined) {
      updates.push('status = ?');
      values.push(dto.status);
    }

    if (updates.length > 0) {
      updates.push('updated_at = CURRENT_TIMESTAMP');
      values.push(id);

      await this.databaseService.run(
        `UPDATE videos SET ${updates.join(', ')} WHERE id = ?`,
        values,
      );
    }

    this.clearCache();
    this.logger.log(`Video updated: ${id}`);

    return this.findById(id);
  }

  // 删除视频
  async remove(id: string): Promise<void> {
    const existing = await this.findById(id);
    if (!existing) {
      throw new NotFoundException('视频不存在');
    }

    await this.databaseService.run('DELETE FROM videos WHERE id = ?', [id]);

    this.clearCache();
    this.logger.log(`Video deleted: ${id}`);
  }

  // 批量排序
  async batchSort(ids: string[]): Promise<void> {
    for (let i = 0; i < ids.length; i++) {
      await this.databaseService.run(
        'UPDATE videos SET "order" = ? WHERE id = ?',
        [i, ids[i]],
      );
    }

    this.clearCache();
    this.logger.log('Videos sorted');
  }

  // 辅助方法
  private async findById(id: string): Promise<Video> {
    const result = await this.databaseService.get<any>(
      'SELECT * FROM videos WHERE id = ?',
      [id],
    );

    if (!result) {
      throw new NotFoundException('视频不存在');
    }

    return this.mapToVideo(result);
  }

  private mapToVideo(row: any): Video {
    return {
      id: row.id,
      title: row.title,
      bvid: row.bvid,
      page: row.page,
      coverUrl: row.cover_url,
      order: row.order,
      status: row.status,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      createdBy: row.created_by,
    };
  }

  private clearCache() {
    this.cacheService.del(this.CACHE_KEY);
    this.cacheService.del(this.CACHE_KEY_ALL);
  }

  private async validateBilibiliVideo(bvid: string): Promise<void> {

}
```

### 6.5 控制器实现

```typescript
// videos.controller.ts
import {
  Controller,
  Get,
  Post,
  Put,
  Delete,
  Body,
  Param,
  Query,
  UseGuards,
  Req,
} from '@nestjs/common';
import { VideosService } from './videos.service';
import { CreateVideoDto } from './dto/create-video.dto';
import { UpdateVideoDto } from './dto/update-video.dto';
import { JwtAuthGuard } from '../auth/guards/jwt-auth.guard';

@Controller()
export class VideosController {
  constructor(private readonly videosService: VideosService) {}

  // 前端接口：获取视频列表
  @Get('videos')
  async getFrontendVideos() {
    return this.videosService.findFrontendVideos();
  }

  // 后台接口：获取视频列表
  @UseGuards(JwtAuthGuard)
  @Get('admin/videos')
  async getBackendVideos(
    @Query('page') page?: number,
    @Query('pageSize') pageSize?: number,
    @Query('keyword') keyword?: string,
    @Query('status') status?: 0 | 1,
  ) {
    return this.videosService.findBackendVideos({
      page: page ? Number(page) : 1,
      pageSize: pageSize ? Number(pageSize) : 20,
      keyword,
      status,
    });
  }

  // 后台接口：创建视频
  @UseGuards(JwtAuthGuard)
  @Post('admin/videos')
  async createVideo(@Body() dto: CreateVideoDto, @Req() req: any) {
    return this.videosService.create(dto, req.user.id);
  }

  // 后台接口：更新视频
  @UseGuards(JwtAuthGuard)
  @Put('admin/videos/:id')
  async updateVideo(
    @Param('id') id: string,
    @Body() dto: UpdateVideoDto,
  ) {
    return this.videosService.update(id, dto);
  }

  // 后台接口：删除视频
  @UseGuards(JwtAuthGuard)
  @Delete('admin/videos/:id')
  async deleteVideo(@Param('id') id: string) {
    return this.videosService.remove(id);
  }

  // 后台接口：批量排序
  @UseGuards(JwtAuthGuard)
  @Put('admin/videos/sort')
  async batchSort(@Body('ids') ids: string[]) {
    return this.videosService.batchSort(ids);
  }
}
```
