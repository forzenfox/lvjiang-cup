#### 2.1.3 核心接口

**1. 下载导入模板**

```
GET /api/admin/teams/import/template
Authorization: Bearer <token>

Response:
  Content-Type: application/vnd.openxmlformats-officedocument.spreadsheetml.sheet
  Content-Disposition: attachment; filename="驴酱杯_战队导入模板_YYYYMMDD.xlsx"
```

**模板生成逻辑**：
- 第一次调用时：检查 `/app/templates/team-import-template.xlsx` 是否存在
  - 若不存在：读取 `lol-champion-map.json`，生成模板，保存到 `/app/templates/team-import-template.xlsx`
  - 若存在：直接读取并返回
- 后续调用：直接返回缓存的模板文件

**2. 批量导入战队**

```
POST /api/admin/teams/import
Authorization: Bearer <token>
Content-Type: multipart/form-data

    "total": 2,
```

```typescript
interface TeamRow {
```

#### 2.1.5 关键实现逻辑

**1. 模板生成逻辑（含缓存）**

```typescript
private readonly TEMPLATE_PATH = '/app/templates/team-import-template.xlsx';
private readonly CHAMPION_MAP_PATH = '/app/data/lol-champion-map.json';

async generateTemplate(): Promise<Buffer> {
  // 1. 检查本地是否存在缓存模板
  if (fs.existsSync(this.TEMPLATE_PATH)) {
    // 返回缓存模板
    return fs.readFileSync(this.TEMPLATE_PATH);
  }

  // 2. 读取英雄数据文件
  const championMap = await this.loadChampionMap();

  // 3. 生成 Sheet 1: 战队与队员信息导入
  const sheet1 = this.createTeamImportSheet();

  // 4. 生成 Sheet 2: 英雄列表（参考）
  const sheet2 = this.createChampionReferenceSheet(championMap);

  // 5. 打包生成 Excel 文件
  const buffer = await this.packageExcel(sheet1, sheet2);

  // 6. 写入缓存文件
  await this.saveTemplateToCache(buffer);

  // 7. 返回 Excel 文件流
  return buffer;
}

async refreshTemplate(): Promise<void> {
  // 删除旧模板
  if (fs.existsSync(this.TEMPLATE_PATH)) {
    fs.unlinkSync(this.TEMPLATE_PATH);
  }
  // 重新生成
  await this.generateTemplate();
}
```

**2. Excel 解析逻辑**

```typescript
async parseExcel(filePath: string): Promise<ImportData[]> {
  const workbook = await exceljs.Workbook.xlsx.readFile(filePath);

  // 验证 Sheet 名称
  const sheet = workbook.getWorksheet('战队与队员信息导入');
  if (!sheet) {
    throw new Error('未找到正确的数据Sheet');
  }

  // 按战队名称分组
  const teams = new Map<string, TeamRow>();
  const members: MemberRow[] = [];

  sheet.eachRow((row, rowNumber) => {
    if (rowNumber <= 3) return; // 跳过标题行

    const teamName = row.getCell(1).value as string;
    const position = row.getCell(4).value as string;

    if (teamName && teamName.trim()) {
      // 战队首行
      teams.set(teamName, {
        team_name: teamName,
        team_logo_url: row.getCell(2).value as string,
        team_battle_cry: row.getCell(3).value as string,
      });
    }

    if (position && POSITION_MAP[position]) {
      // 队员行
      members.push({
        position: POSITION_MAP[position],
        member_nickname: row.getCell(5).value as string,
        member_game_id: row.getCell(6).value as string,
        member_avatar_url: row.getCell(7).value as string,
        member_rating: this.parseRating(row.getCell(8).value),
        member_is_captain: row.getCell(9).value === '是',
        member_level: row.getCell(10).value as string,
        member_champion_pool: this.parseChampionPool(row.getCell(11).value as string),
        member_live_room: this.parseLiveRoom(row.getCell(12).value as string),
        member_bio: row.getCell(13).value as string,
      });
    }
  });

  return this.groupByTeam(teams, members);
}
```

**3. 英雄名称匹配逻辑**

```typescript
// champion-map.json 数据结构
// {
//   "Yasuo": { "id": "Yasuo", "name": "疾风剑豪", "title": "剑豪" },
//   "Ahri": { "id": "Ahri", "name": "九尾妖狐", "title": "阿狸" }
// }

findChampionId(inputName: string): string | null {
  const normalized = inputName.trim().toLowerCase();

  for (const [id, champ] of Object.entries(this.championMap)) {
    if (champ.name === inputName || champ.title === inputName) {
      return id;
    }
  }
  return null;
}

parseChampionPool(poolStr: string): string[] {
  if (!poolStr) return [];

  return poolStr
    .split(',')
    .map(name => name.trim())
    .filter(name => name.length > 0)
    .map(name => this.findChampionId(name))
    .filter(id => id !== null);
}
```

**4. 导入覆盖逻辑**

```typescript
async importTeam(teamData: TeamRow, members: MemberRow[]): Promise<void> {
  // 查找是否已存在同名战队
  const existingTeam = await this.findTeamByName(teamData.team_name);

  if (existingTeam) {
    // 覆盖模式：更新战队信息
    await this.updateTeam(existingTeam.id, {
      name: teamData.team_name,
      logo: teamData.team_logo_url,
      battle_cry: teamData.team_battle_cry,
    });

    // 删除旧队员
    await this.deleteTeamMembers(existingTeam.id);

    // 插入新队员
    for (const member of members) {
      await this.createTeamMember(existingTeam.id, member);
    }
  } else {
    // 新增模式
    const teamId = await this.createTeam(teamData);
    for (const member of members) {
      await this.createTeamMember(teamId, member);
    }
  }
}
```

```typescript
// api/teams-import.ts
import axios from './axios';

export const teamsImportApi = {
  // 下载模板
  downloadTemplate: () => {
    return axios.get('/admin/teams/import/template', {
      responseType: 'blob',
    });
  },

  // 批量导入
  importTeams: (file: File) => {
    const formData = new FormData();
    formData.append('file', file);
    return axios.post('/admin/teams/import', formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
    });
  },

  // 下载错误报告
  downloadErrorReport: (errors: ImportError[]) => {
    return axios.post('/admin/teams/import/error-report', errors, {
      responseType: 'blob',
    });
  },
};
```

#### 2.2.4 组件实现

**ImportDialog 组件**

```tsx
// components/import/ImportDialog.tsx
import { useState } from 'react';
import { teamsImportApi } from '@/api/teams-import';
import { toast } from 'sonner';

export function ImportDialog({ isOpen, onClose, onSuccess }: Props) {
  const [file, setFile] = useState<File | null>(null);
  const [uploading, setUploading] = useState(false);

  const handleImport = async () => {
    if (!file) {
      toast.error('请选择要上传的文件');
      return;
    }

    setUploading(true);
    try {
      const response = await teamsImportApi.importTeams(file);
      const result = response.data;

      if (result.success) {
        toast.success(`导入成功：新增 ${result.data.created} 支战队`);
        onSuccess(result.data);
      } else {
        // 显示错误列表
        onError(result.data.errors);
      }
    } catch (error) {
      toast.error('导入失败，请重试');
    } finally {
      setUploading(false);
    }
  };

  return (
    <Dialog open={isOpen} onOpenChange={onClose}>
      <DialogContent>
        {/* 文件上传区域 */}
        <FileDropzone onFileSelect={setFile} />

        {/* 导入说明 */}
        <div className="text-sm text-gray-500">
          <p>⚠️ 导入说明：</p>
          <ul className="list-disc pl-5">
            <li>队标和头像仅支持外链URL</li>
            <li>导入采用覆盖模式</li>
            <li>战队数量上限为16支</li>
          </ul>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={onClose}>取消</Button>
          <Button onClick={handleImport} disabled={!file || uploading}>
            {uploading ? '导入中...' : '开始导入'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

```typescript
// frontend/tests/e2e/pages/TeamsPage.ts 新增方法

export class TeamsPage {

    const fileChooser = await fileChooserPromise;
    await fileChooser.setFiles(filePath);

    await this.page.getByRole('button', { name: '开始导入' }).click();
  }
```
