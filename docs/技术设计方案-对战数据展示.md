## 3. 数据库设计

### 3.1 新增数据表

#### 3.1.1 match_games 表（对战数据表）

```sql
CREATE TABLE IF NOT EXISTS match_games (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  match_id TEXT NOT NULL,                    -- 关联 matches.id
  game_number INTEGER NOT NULL CHECK(game_number BETWEEN 1 AND 5),  -- 局数 1-5
  winner_team_id TEXT,                       -- 获胜方战队 ID（关联 teams.id）
  game_duration TEXT NOT NULL,               -- 游戏时长，格式 MM:SS
  game_start_time TEXT,                      -- 本局开始时间（ISO 8601）
  blue_team_id TEXT NOT NULL,                -- 蓝色方战队 ID
  red_team_id TEXT NOT NULL,                 -- 红色方战队 ID
  blue_team_kills INTEGER DEFAULT 0,         -- 蓝色方总击杀
  red_team_kills INTEGER DEFAULT 0,          -- 红色方总击杀
  blue_team_gold INTEGER DEFAULT 0,          -- 蓝色方总经济
  red_team_gold INTEGER DEFAULT 0,           -- 红色方总经济
  blue_team_towers INTEGER DEFAULT 0,        -- 蓝色方推塔数
  red_team_towers INTEGER DEFAULT 0,         -- 红色方推塔数
  blue_team_dragons INTEGER DEFAULT 0,       -- 蓝色方控龙数
  red_team_dragons INTEGER DEFAULT 0,        -- 红色方控龙数
  blue_team_barons INTEGER DEFAULT 0,        -- 蓝色方控 Baron 数
  red_team_barons INTEGER DEFAULT 0,         -- 红色方控 Baron 数
  status INTEGER DEFAULT 1 CHECK(status IN (0, 1)),  -- 状态：0-禁用，1-启用
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  created_by TEXT NOT NULL,                  -- 创建人（管理员 ID）
  FOREIGN KEY (match_id) REFERENCES matches(id) ON DELETE CASCADE,
  FOREIGN KEY (winner_team_id) REFERENCES teams(id),
  FOREIGN KEY (blue_team_id) REFERENCES teams(id),
  FOREIGN KEY (red_team_id) REFERENCES teams(id),
  UNIQUE(match_id, game_number)              -- 同一对战的同一局数只能有一条记录
);

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_match_games_match_id ON match_games(match_id);
CREATE INDEX IF NOT EXISTS idx_match_games_status ON match_games(status);
CREATE INDEX IF NOT EXISTS idx_match_games_match_status ON match_games(match_id, status);
```

#### 3.1.2 player_match_stats 表（选手对战数据表）

```sql
CREATE TABLE IF NOT EXISTS player_match_stats (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  match_game_id INTEGER NOT NULL,            -- 关联 match_games.id
  player_id TEXT NOT NULL,                   -- 关联 team_members.id
  team_id TEXT NOT NULL,                     -- 关联 teams.id
  position TEXT NOT NULL CHECK(position IN ('TOP', 'JUNGLE', 'MID', 'ADC', 'SUPPORT')),
  champion_name TEXT NOT NULL,               -- 使用英雄中文名称
  kills INTEGER DEFAULT 0 CHECK(kills >= 0),
  deaths INTEGER DEFAULT 0 CHECK(deaths >= 0),
  assists INTEGER DEFAULT 0 CHECK(assists >= 0),
  cs INTEGER DEFAULT 0 CHECK(cs >= 0),       -- 补刀数
  gold INTEGER DEFAULT 0 CHECK(gold >= 0),   -- 总经济
  damage_dealt INTEGER DEFAULT 0 CHECK(damage_dealt >= 0),  -- 造成伤害
  damage_taken INTEGER DEFAULT 0 CHECK(damage_taken >= 0),  -- 承受伤害
  vision_score INTEGER DEFAULT 0,            -- 视野得分
  wards_placed INTEGER DEFAULT 0,            -- 插眼数
  level INTEGER DEFAULT 1,                   -- 最终等级
  first_blood INTEGER DEFAULT 0 CHECK(first_blood IN (0, 1)),  -- 是否一血
  mvp INTEGER DEFAULT 0 CHECK(mvp IN (0, 1)),  -- 是否 MVP
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (match_game_id) REFERENCES match_games(id) ON DELETE CASCADE,
  FOREIGN KEY (player_id) REFERENCES team_members(id),
  FOREIGN KEY (team_id) REFERENCES teams(id),
  UNIQUE(match_game_id, player_id)           -- 同一局中同一选手只能有一条记录
);

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_player_stats_match_game ON player_match_stats(match_game_id);
CREATE INDEX IF NOT EXISTS idx_player_stats_player ON player_match_stats(player_id);
CREATE INDEX IF NOT EXISTS idx_player_stats_team ON player_match_stats(team_id);
CREATE INDEX IF NOT EXISTS idx_player_stats_position ON player_match_stats(position);
```

```typescript
async checkMatchDataExists(matchId: string) {
  const cacheKey = `match_data:check:${matchId}`;
  
  return this.cacheService.getOrSet(cacheKey, async () => {
    const result = await get(
      this.db,
      `SELECT COUNT(*) as count FROM match_games 
       WHERE match_id = ? AND status = 1`,
      [matchId]
    );
    
    return {
      hasData: result.count > 0,
      gameCount: result.count
    };
  }, 60); // 缓存 60 秒
}
```

```typescript
async getMatchSeries(matchId: string) {
  const cacheKey = `match_data:series:${matchId}`;
  
  return this.cacheService.getOrSet(cacheKey, async () => {
    // 获取比赛基本信息
    const match = await get(
      this.db,
      `SELECT id, team_a_id, team_b_id, bo_format 
       FROM matches WHERE id = ?`,
      [matchId]
    );
    
    if (!match) {
      throw new NotFoundException('比赛不存在');
    }
    
    // 获取战队信息
    const teamA = await get(this.db, `SELECT id, name FROM teams WHERE id = ?`, [match.team_a_id]);
    const teamB = await get(this.db, `SELECT id, name FROM teams WHERE id = ?`, [match.team_b_id]);
    
    // 获取所有局数信息
    const games = await all(
      this.db,
      `SELECT game_number, winner_team_id, game_duration, status
       FROM match_games 
       WHERE match_id = ? 
       ORDER BY game_number ASC`,
      [matchId]
    );
    
    // 构建完整局数列表（根据赛制）
    const maxGames = match.bo_format === 'BO1' ? 1 : match.bo_format === 'BO3' ? 3 : 5;
    const gameMap = new Map(games.map(g => [g.game_number, g]));
    
    const gamesList = Array.from({ length: maxGames }, (_, i) => {
      const game = gameMap.get(i + 1);
      return {
        gameNumber: i + 1,
        winnerTeamId: game?.winner_team_id || null,
        gameDuration: game?.game_duration || null,
        hasData: game?.status === 1
      };
    });
    
    return {
      matchId: match.id,
      teamA,
      teamB,
      format: match.bo_format,
      games: gamesList
    };
  }, 60);
}
```

**Service 实现要点：**
```typescript
async getMatchGameData(matchId: string, gameNumber: number) {
  const cacheKey = `match_data:game:${matchId}:${gameNumber}`;
  
  return this.cacheService.getOrSet(cacheKey, async () => {
    // 验证局数
    const match = await get(this.db, `SELECT bo_format FROM matches WHERE id = ?`, [matchId]);
    const maxGames = match.bo_format === 'BO1' ? 1 : match.bo_format === 'BO3' ? 3 : 5;
    if (gameNumber < 1 || gameNumber > maxGames) {
      throw new BadRequestException('局数超出赛制限制', { code: 40002 });
    }
    
    // 获取对战数据
    const game = await get(
      this.db,
      `SELECT * FROM match_games WHERE match_id = ? AND game_number = ? AND status = 1`,
      [matchId, gameNumber]
    );
    
    if (!game) {
      throw new NotFoundException('对战数据不存在', { code: 40003 });
    }
    
    // 获取战队信息
    const blueTeam = await get(this.db, `SELECT id, name FROM teams WHERE id = ?`, [game.blue_team_id]);
    const redTeam = await get(this.db, `SELECT id, name FROM teams WHERE id = ?`, [game.red_team_id]);
    
    // 获取选手数据
    const playerStats = await all(
      this.db,
      `SELECT 
         ps.*, 
         tm.nickname as player_name,
         t.name as team_name
       FROM player_match_stats ps
       LEFT JOIN team_members tm ON ps.player_id = tm.id
       LEFT JOIN teams t ON ps.team_id = t.id
       WHERE ps.match_game_id = ?
       ORDER BY 
         CASE ps.position
           WHEN 'TOP' THEN 1
           WHEN 'JUNGLE' THEN 2
           WHEN 'MID' THEN 3
           WHEN 'ADC' THEN 4
           WHEN 'SUPPORT' THEN 5
         END`,
      [game.id]
    );
    
    // 计算 KDA 字符串
    const playerStatsWithKda = playerStats.map(ps => ({
      ...ps,
      kda: `${ps.kills}/${ps.deaths}/${ps.assists}`
    }));
    
    return {
      id: game.id,
      matchId: game.match_id,
      gameNumber: game.game_number,
      winnerTeamId: game.winner_team_id,
      gameDuration: game.game_duration,
      gameStartTime: game.game_start_time,
      blueTeam: {
        teamId: blueTeam.id,
        teamName: blueTeam.name,
        side: 'blue',
        kills: game.blue_team_kills,
        gold: game.blue_team_gold,
        towers: game.blue_team_towers,
        dragons: game.blue_team_dragons,
        barons: game.blue_team_barons,
        isWinner: game.winner_team_id === blueTeam.id
      },
      redTeam: {
        teamId: redTeam.id,
```

```typescript
// 计算玩家雷达图数值
function getPlayerRadarValue(player: PlayerStat, key: string): number {
  const gameDurationMinutes = parseGameDuration(player.gameDuration);
  
  switch (key) {
    case 'csPerMin':
      return player.cs / gameDurationMinutes;
    case 'damageShare':
      return (player.damageDealt / player.teamTotalDamage) * 100;
    case 'damageTakenShare':
      return (player.damageTaken / player.teamTotalDamageTaken) * 100;
    case 'killParticipation':
      return ((player.kills + player.assists) / player.teamTotalKills) * 100;
    case 'damagePerGold':
      return (player.damageDealt / player.gold) * 100;
    case 'kda':
      return player.deaths === 0 
        ? (player.kills + player.assists) 
        : (player.kills + player.assists) / player.deaths;
    case 'goldPerMin':
      return player.gold / gameDurationMinutes;
    case 'damagePerMin':
      return player.damageDealt / gameDurationMinutes;
    case 'wardsPerMin':
      return player.wardsPlaced / gameDurationMinutes;
    case 'damageTakenPerDeath':
      return player.deaths === 0 ? player.damageTaken : player.damageTaken / player.deaths;
    case 'avgAssists':
      return player.assists;
    default:
      return 0;
  }
}

// 解析游戏时长（MM:SS → 分钟）
function parseGameDuration(duration: string): number {
  const [minutes, seconds] = duration.split(':').map(Number);
  return minutes + seconds / 60;
}
```

### 6.1 Zustand Store 设计

```typescript
// src/store/matchDataStore.ts

import { create } from 'zustand';
import { getMatchGameData, getMatchSeries, checkMatchDataExists } from '@/api/matchData';
import type { MatchGameData, MatchSeriesInfo } from '@/types/matchData';

interface MatchDataState {
  // 数据状态
  gameData: MatchGameData | null;
  seriesInfo: MatchSeriesInfo | null;
  hasData: boolean;
  
  // 加载状态
  loading: boolean;
  error: string | null;
  
  // UI 状态
  selectedGameNumber: number;
  expandedPlayerId: number | null;
  
  // Actions
  fetchGameData: (matchId: string, gameNumber: number) => Promise<void>;
  fetchSeriesInfo: (matchId: string) => Promise<void>;
  checkHasData: (matchId: string) => Promise<void>;
  setSelectedGame: (gameNumber: number) => void;
  setExpandedPlayer: (playerId: number | null) => void;
  reset: () => void;
}

export const useMatchDataStore = create<MatchDataState>((set, get) => ({
  // 初始状态
  gameData: null,
  seriesInfo: null,
  hasData: false,
  loading: false,
  error: null,
  selectedGameNumber: 1,
  expandedPlayerId: null,
  
  // 获取对战数据详情
  fetchGameData: async (matchId: string, gameNumber: number) => {
    set({ loading: true, error: null });
    try {
      const response = await getMatchGameData(matchId, gameNumber);
      if (response.success) {
        set({ 
          gameData: response.data,
          selectedGameNumber: gameNumber,
          loading: false 
        });
      } else {
        set({ error: response.message || '加载失败', loading: false });
      }
    } catch (err) {
      set({ error: err.message || '网络错误', loading: false });
    }
  },
  
  // 获取系列赛信息
  fetchSeriesInfo: async (matchId: string) => {
    try {
      const response = await getMatchSeries(matchId);
      if (response.success) {
        set({ seriesInfo: response.data });
      }
    } catch (err) {
      console.error('Failed to fetch series info:', err);
    }
  },
  
  // 检查数据是否存在
  checkHasData: async (matchId: string) => {
    try {
      const response = await checkMatchDataExists(matchId);
      if (response.success) {
        set({ hasData: response.data.hasData });
      }
    } catch (err) {
      console.error('Failed to check data:', err);
    }
  },
  
  // 设置当前局数
  setSelectedGame: (gameNumber: number) => {
    set({ selectedGameNumber: gameNumber });
  },
  
  // 设置展开的选手
  setExpandedPlayer: (playerId: number | null) => {
    set({ expandedPlayerId: playerId });
  },
  
  // 重置状态
  reset: () => {
    set({
      gameData: null,
      seriesInfo: null,
      hasData: false,
      loading: false,
      error: null,
      selectedGameNumber: 1,
      expandedPlayerId: null
    });
  }
}));
```

### 6.2 缓存策略

**前端缓存：**
- 使用 Zustand 状态管理，组件卸载时自动释放
- 对于频繁访问的数据（系列信息），使用 localStorage 缓存 5 分钟
- 对局切换时预加载相邻局数据

**后端缓存：**
- 使用 node-cache，TTL 60 秒
- 缓存键格式：
  - `match_data:check:{matchId}` - 数据存在检查
  - `match_data:series:{matchId}` - 系列赛信息
  - `match_data:game:{matchId}:{gameNumber}` - 对战数据详情
- 数据更新时清除相关缓存

```typescript
// backend/src/modules/utils/excel.util.ts

import * as XLSX from 'xlsx';
import type { MatchGameData, PlayerStat } from '@/types/matchData';

interface ParsedMatchData {
  matchInfo: MatchInfo;
  teamStats: TeamStats[];
  playerStats: PlayerStat[];
}

interface MatchInfo {
  gameNumber: number;
  winner: string;
  gameDuration: string;
  gameStartTime: string;
}

interface TeamStats {
  teamName: string;
  side: 'red' | 'blue';
  kills: number;
  gold: number;
  towers: number;
  dragons: number;
  barons: number;
}

export function parseMatchDataExcel(buffer: Buffer): ParsedMatchData {
  // 读取 Excel 文件
  const workbook = XLSX.read(buffer, { type: 'buffer' });
  const sheetName = workbook.SheetNames[0];
  const worksheet = workbook.Sheets[sheetName];
  
  // 转换为二维数组
  const data = XLSX.utils.sheet_to_json(worksheet, { header: 1 }) as any[][];
  
  // 验证行数（应为 16 行）
  if (data.length !== 16) {
    throw new Error('Excel 文件格式错误：应为 16 行数据');
  }
  
  // 解析 MatchInfo（第 2 行，索引 1）
  const matchInfoRow = data[1];
  const matchInfo: MatchInfo = {
    gameNumber: matchInfoRow[0],
    winner: matchInfoRow[1],
    gameDuration: matchInfoRow[2],
    gameStartTime: matchInfoRow[3]
  };
  
  // 验证游戏时长格式（MM:SS）
  if (!/^\d{1,2}:\d{2}$/.test(matchInfo.gameDuration)) {
    throw new Error('游戏时长格式错误，应为 MM:SS');
  }
  
  // 解析 TeamStats（第 4-5 行，索引 3-4）
  const teamStats: TeamStats[] = [
    {
      teamName: data[3][0],
      side: 'red',
      kills: data[3][1],
      gold: data[3][2],
      towers: data[3][3],
      dragons: data[3][4],
      barons: data[3][5]
    },
    {
      teamName: data[4][0],
      side: 'blue',
      kills: data[4][1],
      gold: data[4][2],
      towers: data[4][3],
      dragons: data[4][4],
      barons: data[4][5]
    }
  ];
  
  // 验证战队名称
  for (const team of teamStats) {
    if (!team.teamName) {
      throw new Error('战队名称不能为空');
    }
  }
  
  // 解析 PlayerStats（第 7-16 行，索引 6-15）
  const playerStats: PlayerStat[] = [];
  for (let i = 6; i <= 15; i++) {
    const row = data[i];
    
    // 验证必填字段
    if (!row[0] || !row[1] || !row[2]) {
      throw new Error(`第 ${i + 1} 行数据不完整`);
    }
    
    playerStats.push({
      teamName: row[0],
      playerName: row[1],
      position: row[2],
      championName: row[3],
      kills: row[4] || 0,
      deaths: row[5] || 0,
      assists: row[6] || 0,
      cs: row[7] || 0,
      gold: row[8] || 0,
      damageDealt: row[9] || 0,
      damageTaken: row[10] || 0,
      visionScore: row[11] || 0,
      wardsPlaced: row[12] || 0,
      level: row[13] || 1,
      firstBlood: row[14] === '是' ? 1 : 0,
      mvp: row[15] === '是' ? 1 : 0
    });
  }
  
  // 验证选手数据（应为 10 人，每队 5 人）
  if (playerStats.length !== 10) {
    throw new Error('选手数据应为 10 人');
  }
  
  // 验证位置
  const validPositions = ['TOP', 'JUNGLE', 'MID', 'ADC', 'SUPPORT'];
  for (const player of playerStats) {
    if (!validPositions.includes(player.position)) {
      throw new Error(`位置 ${player.position} 无效`);
    }
  }
  
  // 验证 KDA 非负
  for (const player of playerStats) {
    if (player.kills < 0 || player.deaths < 0 || player.assists < 0) {
      throw new Error('KDA 数值不能为负数');
    }
  }
  
  return { matchInfo, teamStats, playerStats };
}
```

```typescript
// backend/src/modules/match-data/match-data.service.ts

async importMatchData(matchId: string, file: Express.Multer.File, adminId: string) {
  // 1. 解析 Excel 文件
  const parsedData = parseMatchDataExcel(file.buffer);
  
  // 2. 验证比赛
  const match = await get(this.db, `SELECT id, bo_format, team_a_id, team_b_id FROM matches WHERE id = ?`, [matchId]);
  if (!match) {
    throw new NotFoundException('比赛不存在');
  }
  
  // 3. 验证局数
  const maxGames = match.bo_format === 'BO1' ? 1 : match.bo_format === 'BO3' ? 3 : 5;
  if (parsedData.matchInfo.gameNumber < 1 || parsedData.matchInfo.gameNumber > maxGames) {
    throw new BadRequestException('局数超出赛制限制', { code: 40002 });
  }
  
  // 4. 战队名称匹配战队 ID
  const redTeam = await get(this.db, `SELECT id FROM teams WHERE name = ?`, [parsedData.teamStats[0].teamName]);
  const blueTeam = await get(this.db, `SELECT id FROM teams WHERE name = ?`, [parsedData.teamStats[1].teamName]);
  
  if (!redTeam || !blueTeam) {
    throw new BadRequestException('战队名称不匹配');
  }
  
  // 5. 开始事务
  await this.dbService.beginTransaction();
  
  try {
    // 6. 插入/更新对战数据
    const existingGame = await get(
      this.db,
      `SELECT id FROM match_games WHERE match_id = ? AND game_number = ?`,
      [matchId, parsedData.matchInfo.gameNumber]
    );
    
    let gameId: number;
    
    if (existingGame) {
      // 更新现有数据
      await run(
        this.db,
        `UPDATE match_games SET 
          winner_team_id = ?, 
          game_duration = ?, 
          game_start_time = ?, 
          blue_team_id = ?, 
          red_team_id = ?, 
          blue_team_kills = ?, 
          red_team_kills = ?, 
          blue_team_gold = ?, 
          red_team_gold = ?, 
          blue_team_towers = ?, 
          red_team_towers = ?, 
          blue_team_dragons = ?, 
          red_team_dragons = ?, 
          blue_team_barons = ?, 
          red_team_barons = ?, 
          updated_at = CURRENT_TIMESTAMP
        WHERE id = ?`,
        [
          parsedData.matchInfo.winner === blueTeam.name ? blueTeam.id : redTeam.id,
          parsedData.matchInfo.gameDuration,
          parsedData.matchInfo.gameStartTime,
          blueTeam.id,
          redTeam.id,
          parsedData.teamStats[1].kills,
          parsedData.teamStats[0].kills,
          parsedData.teamStats[1].gold,
          parsedData.teamStats[0].gold,
          parsedData.teamStats[1].towers,
          parsedData.teamStats[0].towers,
          parsedData.teamStats[1].dragons,
          parsedData.teamStats[0].dragons,
          parsedData.teamStats[1].barons,
          parsedData.teamStats[0].barons,
          existingGame.id
        ]
      );
      gameId = existingGame.id;
    } else {
      // 插入新数据
      const result = await run(
        this.db,
        `INSERT INTO match_games (
          match_id, game_number, winner_team_id, game_duration, game_start_time,
          blue_team_id, red_team_id, 
          blue_team_kills, red_team_kills, blue_team_gold, red_team_gold,
          blue_team_towers, red_team_towers, blue_team_dragons, red_team_dragons,
          blue_team_barons, red_team_barons, created_by
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
        [
          matchId,
          parsedData.matchInfo.gameNumber,
          parsedData.matchInfo.winner === blueTeam.name ? blueTeam.id : redTeam.id,
          parsedData.matchInfo.gameDuration,
          parsedData.matchInfo.gameStartTime,
          blueTeam.id,
          redTeam.id,
          parsedData.teamStats[1].kills,
          parsedData.teamStats[0].kills,
          parsedData.teamStats[1].gold,
          parsedData.teamStats[0].gold,
          parsedData.teamStats[1].towers,
          parsedData.teamStats[0].towers,
          parsedData.teamStats[1].dragons,
          parsedData.teamStats[0].dragons,
          parsedData.teamStats[1].barons,
          parsedData.teamStats[0].barons,
          adminId
        ]
      );
      gameId = result.lastID;
    }
    
    // 7. 删除旧的选手数据
    await run(this.db, `DELETE FROM player_match_stats WHERE match_game_id = ?`, [gameId]);
    
    // 8. 插入选手数据
    for (const player of parsedData.playerStats) {
      // 选手昵称匹配
      const teamId = player.teamName === blueTeam.name ? blueTeam.id : redTeam.id;
      const member = await get(
        this.db,
        `SELECT id FROM team_members WHERE team_id = ? AND LOWER(TRIM(nickname)) = LOWER(TRIM(?))`,
        [teamId, player.playerName]
      );
      
      if (!member) {
        throw new BadRequestException(`选手 ${player.playerName} 不存在于战队 ${player.teamName}`);
      }
      
      await run(
        this.db,
        `INSERT INTO player_match_stats (
          match_game_id, player_id, team_id, position, champion_name,
          kills, deaths, assists, cs, gold, damage_dealt, damage_taken,
          vision_score, wards_placed, level, first_blood, mvp
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
        [
          gameId,
          member.id,
          teamId,
          player.position,
          player.championName,
          player.kills,
          player.deaths,
          player.assists,
          player.cs,
          player.gold,
          player.damageDealt,
          player.damageTaken,
          player.visionScore,
          player.wardsPlaced,
          player.level,
          player.firstBlood,
          player.mvp
        ]
      );
    }
    
    // 9. 提交事务
    await this.dbService.commitTransaction();
    
    // 10. 清除缓存
    await this.cacheService.del(`match_data:game:${matchId}:${parsedData.matchInfo.gameNumber}`);
    await this.cacheService.del(`match_data:series:${matchId}`);
    await this.cacheService.del(`match_data:check:${matchId}`);
    
    return {
      imported: true,
      gameNumber: parsedData.matchInfo.gameNumber,
      playerCount: parsedData.playerStats.length
    };
    
  } catch (error) {
    // 11. 回滚事务
    await this.dbService.rollbackTransaction();
    throw error;
  }
}
```

### 14.2 英雄名称映射表

```typescript
// src/utils/championMap.ts
export const championNameMap: Record<string, string> = {
  '亚索': '亚索',
  '劫': '劫',
  ' EZ': '伊泽瑞尔',
  // ... 其他英雄映射
};

// Excel 导入时使用模糊匹配
function matchChampionName(input: string): string | null {
  const normalizedInput = input.trim().toLowerCase();
  for (const [key, value] of Object.entries(championNameMap)) {
    if (key.toLowerCase() === normalizedInput) {
      return value;
    }
  }
  return null;
}
```

### 14.3 错误处理最佳实践

```typescript
// 统一错误响应格式
class AppException extends Error {
  constructor(
    message: string,
    public code: number,
    public statusCode: number = 400
  ) {
    super(message);
  }
}

// 全局异常过滤器
// backend/src/common/filters/global-exception.filter.ts
import { ExceptionFilter, Catch, ArgumentsHost, HttpException } from '@nestjs/common';

@Catch()
export class GlobalExceptionFilter implements ExceptionFilter {
  catch(exception: unknown, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    
    if (exception instanceof AppException) {
      response.status(exception.statusCode).json({
        success: false,
        code: exception.code,
        message: exception.message
      });
    } else if (exception instanceof HttpException) {
      response.status(exception.getStatus()).json({
        success: false,
        code: exception.getStatus(),
        message: exception.message
      });
    } else {
      // 未知错误
      response.status(500).json({
        success: false,
        code: 50000,
        message: '服务器内部错误'
      });
    }
  }
}
```
